/// model. The handler computes the post-toggle URL once per chip; the
/// template renders it via `<a href="?{{ chip.href }}" hx-get="?{{ chip.href }}">`.
/// Single source of truth for href + hx-get (DRY); avoids inline askama

    /// `filter=&sort=name&order=asc&tag=backup&tag=weekly`. Active tags

**Step B — Add helper function** to compute chips. Place near other private helpers (after `to_view` at ~L173-200):

```rust
/// Phase 23: build a Vec<ChipView> from the fleet-tag set + active set
/// + current request params. Each chip's href is the post-toggle URL —
/// active chips emit URLs that REMOVE their tag; inactive chips emit
/// URLs that ADD their tag. The active set is canonicalized alphabetical
/// before serialization so all chip URLs (and bookmarks) share the same
/// canonical form.
fn build_chip_views(
    fleet_tags: &[String],
    active_tags: &[String],
    filter: &str,
    sort: &str,
    order: &str,
) -> Vec<ChipView> {
    fleet_tags
        .iter()
        .map(|tag| {
            let is_active = active_tags.iter().any(|t| t == tag);

            // Compute the post-toggle active set:
            //  - if currently active → REMOVE this tag (toggle off)
            //  - if currently inactive → ADD this tag (toggle on)
            let mut next: Vec<String> = active_tags.iter().cloned().collect();
            if is_active {
                next.retain(|t| t != tag);
            } else {
                next.push(tag.clone());
                next.sort();
                next.dedup();
            }

            // Build the query string. Tags are percent-encoded via
            // urlencoding::encode for defense-in-depth (charset regex
            // already prevents structural escape but never trust input
            // round-tripping).
            let mut href = format!(
                "filter={}&sort={}&order={}",
                urlencoding::encode(filter),
                urlencoding::encode(sort),
                urlencoding::encode(order)
            );
            for t in &next {
                href.push_str(&format!("&tag={}", urlencoding::encode(t)));
            }

            let aria_label = if is_active {
                format!("Tag filter: {} (active — click to remove)", tag)
            } else {
                format!("Tag filter: {} (inactive — click to add)", tag)
            };

            ChipView {
                tag: tag.clone(),
                is_active,
                href,
                aria_label,
            }
        })
        .collect()
}
```
