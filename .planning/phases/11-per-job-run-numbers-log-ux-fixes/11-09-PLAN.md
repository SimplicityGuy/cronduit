```rust
async fn fetch_logs(
    pool: &DbPool,
    run_id: i64,
    offset: i64,
) -> (Vec<LogLineView>, i64, bool, i64, i64) {
    let page = queries::get_log_lines(pool, run_id, LOG_PAGE_SIZE, offset)
        .await
        .unwrap_or(Paginated { items: Vec::new(), total: 0 });
    let logs: Vec<LogLineView> = page.items
        .into_iter()
        .map(|db_log| LogLineView {
            id: db_log.id,
            is_stderr: db_log.stream == "stderr",
            stream: db_log.stream,
            ts: db_log.ts,
            html: html_escape::encode_text(&db_log.line).into_owned(),
        })
        .collect();
    let total = page.total;
    let has_older = offset + (logs.len() as i64) < total;
    let next_offset = offset + logs.len() as i64;
    let last_log_id = logs.iter().map(|l| l.id).max().unwrap_or(0);
    (logs, total, has_older, next_offset, last_log_id)
}
```
