```rust
pub async fn insert_log_batch(
    pool: &DbPool,
    run_id: i64,
    lines: &[(String, String, String)],
) -> anyhow::Result<Vec<i64>> {
    if lines.is_empty() {
        return Ok(Vec::new());
    }
    let mut ids = Vec::with_capacity(lines.len());

    match pool.writer() {
        PoolRef::Sqlite(p) => {
            let mut tx = p.begin().await?;
            for (stream, ts, line) in lines {
                let id: i64 = sqlx::query_scalar(
                    "INSERT INTO job_logs (run_id, stream, ts, line) \
                     VALUES (?1, ?2, ?3, ?4) RETURNING id",
                )
                .bind(run_id)
                .bind(stream)
                .bind(ts)
                .bind(line)
                .fetch_one(&mut *tx)
                .await?;
                ids.push(id);
            }
            tx.commit().await?;
        }
        PoolRef::Postgres(p) => {
            let mut tx = p.begin().await?;
            for (stream, ts, line) in lines {
                let id: i64 = sqlx::query_scalar(
                    "INSERT INTO job_logs (run_id, stream, ts, line) \
                     VALUES ($1, $2, $3, $4) RETURNING id",
                )
                .bind(run_id)
                .bind(stream)
                .bind(ts)
                .bind(line)
                .fetch_one(&mut *tx)
                .await?;
                ids.push(id);
            }
            tx.commit().await?;
        }
    }
    Ok(ids)
}
```

/// T-V11-LOG-01: broadcast lines must carry id populated.
///
/// Rather than spinning up the full scheduler's log_writer_task (which owns the
/// broadcast_tx internally), we reproduce its zip-with-ids contract here: call
/// insert_log_batch, get the Vec<i64>, then zip with the input batch and send
/// each LogLine through a test-local broadcast::channel. Assert on the receiver
/// side that every received LogLine has id == Some(_) AND those Some values
/// equal the Vec<i64> from insert_log_batch. This locks the DB → broadcast
/// contract that log_writer_task implements in production (Task 3).
#[tokio::test]
async fn broadcast_id_populated() {
    let pool = setup_sqlite_with_phase11_migrations().await;
    let job_id = seed_test_job(&pool, "broadcast-id-job").await;
    let run_id = seed_running_run(&pool, job_id).await;

    // Stand up a broadcast channel + subscribe a receiver BEFORE sending.
    let (broadcast_tx, mut broadcast_rx) = broadcast::channel::<LogLine>(64);

    // Produce a small batch of 3 lines.
    let batch = make_test_batch(3);

    // Persist via insert_log_batch and capture the ids.
    let ids = insert_log_batch(&pool, run_id, &batch)
        .await
        .expect("insert batch");
    assert_eq!(ids.len(), 3, "setup: one id per input line");

    // Mirror the log_writer_task zip step (Task 3 of this plan): construct
    // LogLine { id: Some(..), stream, ts, line } from batch + ids and send.
    let lines: Vec<LogLine> = batch
        .iter()
        .map(|(stream, ts, line)| LogLine {
            id: None, // pre-broadcast value; replaced below via zip
            stream: stream.clone(),
            ts: ts.clone(),
            line: line.clone(),
        })
        .collect();
    for (line, id) in lines.into_iter().zip(ids.clone().into_iter()) {
        broadcast_tx
            .send(LogLine { id: Some(id), ..line })
            .expect("send to subscribed receiver");
    }
    drop(broadcast_tx);

    // Drain the receiver — must get exactly 3 LogLines, each with id.is_some()
    // AND the collected ids must equal the Vec<i64> from insert_log_batch.
    let mut received_ids: Vec<i64> = Vec::new();
    while let Ok(line) = broadcast_rx.recv().await {
        assert!(
            line.id.is_some(),
            "every broadcast LogLine MUST carry id = Some(_) per D-01: {:?}",
            line
        );
        received_ids.push(line.id.unwrap());
    }
    assert_eq!(
        received_ids, ids,
        "broadcast ids must match insert_log_batch's Vec<i64> in order"
    );
}
