<objective>
Implement the daily retention pruner that deletes old job_runs and job_logs in batched transactions.

Purpose: Without retention, the SQLite database grows unboundedly. The pruner keeps storage predictable by removing data older than [server].log_retention (default 90 days) while avoiding write contention (DB-08, Pitfall 11).

Output: A background tokio task that runs on a 24-hour interval, deletes logs then runs in batches of 1000, issues WAL checkpoint after large prunes, and shuts down cleanly.
</objective>

   /// Delete a batch of job_logs rows where the associated run ended before the cutoff.
   /// Returns the number of rows deleted. Deletes logs BEFORE runs (FK safety).
   pub async fn delete_old_logs_batch(pool: &DbPool, cutoff: &str, batch_size: i64) -> Result<i64, sqlx::Error> {
       match pool {
           DbPool::Sqlite { write, .. } => {
               let result = sqlx::query(
                   "DELETE FROM job_logs WHERE rowid IN (
                       SELECT jl.rowid FROM job_logs jl
                       INNER JOIN job_runs jr ON jl.run_id = jr.id
                       WHERE jr.end_time IS NOT NULL AND jr.end_time < ?1
                       LIMIT ?2
                   )"
               )
               .bind(cutoff)
               .bind(batch_size)
               .execute(write)
               .await?;
               Ok(result.rows_affected() as i64)
           }
           DbPool::Postgres(pg) => {
               let result = sqlx::query(
                   "DELETE FROM job_logs WHERE id IN (
                       SELECT jl.id FROM job_logs jl
                       INNER JOIN job_runs jr ON jl.run_id = jr.id
                       WHERE jr.end_time IS NOT NULL AND jr.end_time < $1
                       LIMIT $2
                   )"
               )
               .bind(cutoff)
               .bind(batch_size)
               .execute(pg)
               .await?;
               Ok(result.rows_affected() as i64)
           }
       }
   }

   /// Delete a batch of job_runs rows that ended before the cutoff and have no remaining logs.
   pub async fn delete_old_runs_batch(pool: &DbPool, cutoff: &str, batch_size: i64) -> Result<i64, sqlx::Error> {
       match pool {
           DbPool::Sqlite { write, .. } => {
               let result = sqlx::query(
                   "DELETE FROM job_runs WHERE rowid IN (
                       SELECT jr.rowid FROM job_runs jr
                       WHERE jr.end_time IS NOT NULL AND jr.end_time < ?1
                       AND NOT EXISTS (SELECT 1 FROM job_logs jl WHERE jl.run_id = jr.id)
                       LIMIT ?2
                   )"
               )
               .bind(cutoff)
               .bind(batch_size)
               .execute(write)
               .await?;
               Ok(result.rows_affected() as i64)
           }
           DbPool::Postgres(pg) => {
               let result = sqlx::query(
                   "DELETE FROM job_runs WHERE id IN (
                       SELECT jr.id FROM job_runs jr
                       WHERE jr.end_time IS NOT NULL AND jr.end_time < $1
                       AND NOT EXISTS (SELECT 1 FROM job_logs jl WHERE jl.run_id = jr.id)
                       LIMIT $2
                   )"
               )
               .bind(cutoff)
               .bind(batch_size)
               .execute(pg)
               .await?;
               Ok(result.rows_affected() as i64)
           }
       }
   }

   /// Issue WAL checkpoint on SQLite to reclaim space after large deletes.
   pub async fn wal_checkpoint(pool: &DbPool) -> Result<(), sqlx::Error> {
       match pool {
           DbPool::Sqlite { write, .. } => {
               sqlx::query("PRAGMA wal_checkpoint(TRUNCATE)")
                   .execute(write)
                   .await?;
               Ok(())
           }
           DbPool::Postgres(_) => Ok(()), // Postgres auto-vacuums
       }
   }
   ```

   //! Daily retention pruner (DB-08).
   //!
   //! Deletes job_logs and job_runs older than [server].log_retention in batches
   //! to avoid SQLite write contention (Pitfall 11).

   use std::time::Duration;
   use tokio_util::sync::CancellationToken;
   use crate::db::{DbPool, queries};

   const BATCH_SIZE: i64 = 1000;
   const BATCH_SLEEP: Duration = Duration::from_millis(100);
   const WAL_CHECKPOINT_THRESHOLD: i64 = 10_000;

   /// Spawn the retention pruner as a background task.
   /// Runs on a 24-hour interval from startup (not wall-clock aligned).
   /// Checks CancellationToken between batches for graceful shutdown.
   pub async fn retention_pruner(
       pool: DbPool,
       retention: Duration,
       cancel: CancellationToken,
   ) {
       let mut interval = tokio::time::interval(Duration::from_secs(86400));
       // Skip the initial immediate tick -- first prune after 24 hours
       interval.tick().await;

       loop {
           tokio::select! {
               _ = interval.tick() => {
                   run_prune_cycle(&pool, retention, &cancel).await;
               }
               _ = cancel.cancelled() => {
                   tracing::info!("retention_pruner shutting down");
                   break;
               }
           }
       }
   }

   async fn run_prune_cycle(pool: &DbPool, retention: Duration, cancel: &CancellationToken) {
       let cutoff = chrono::Utc::now()
           - chrono::Duration::from_std(retention).unwrap_or(chrono::Duration::days(90));
       let cutoff_str = cutoff.to_rfc3339();

       tracing::info!(cutoff = %cutoff_str, "retention prune cycle started");

       // Phase 1: Delete old log lines (children first for FK safety)
       let mut total_logs_deleted: i64 = 0;
       loop {
           if cancel.is_cancelled() {
               tracing::warn!(logs_deleted = total_logs_deleted, "prune interrupted by shutdown");
               return;
           }
           match queries::delete_old_logs_batch(pool, &cutoff_str, BATCH_SIZE).await {
               Ok(deleted) => {
                   total_logs_deleted += deleted;
                   if deleted > 0 {
                       tracing::debug!(deleted, total = total_logs_deleted, "prune_batch: logs");
                   }
                   if deleted < BATCH_SIZE {
                       break;
                   }
                   tokio::time::sleep(BATCH_SLEEP).await;
               }
               Err(e) => {
                   tracing::error!(error = %e, "retention prune: failed to delete log batch");
                   break;
               }
           }
       }

       // Phase 2: Delete orphaned job_runs (no remaining logs)
       let mut total_runs_deleted: i64 = 0;
       loop {
           if cancel.is_cancelled() {
               tracing::warn!(
                   logs_deleted = total_logs_deleted,
                   runs_deleted = total_runs_deleted,
                   "prune interrupted by shutdown"
               );
               return;
           }
           match queries::delete_old_runs_batch(pool, &cutoff_str, BATCH_SIZE).await {
               Ok(deleted) => {
                   total_runs_deleted += deleted;
                   if deleted > 0 {
                       tracing::debug!(deleted, total = total_runs_deleted, "prune_batch: runs");
                   }
                   if deleted < BATCH_SIZE {
                       break;
                   }
                   tokio::time::sleep(BATCH_SLEEP).await;
               }
               Err(e) => {
                   tracing::error!(error = %e, "retention prune: failed to delete run batch");
                   break;
               }
           }
       }

       // Phase 3: WAL checkpoint if large prune (SQLite only)
       let total_deleted = total_logs_deleted + total_runs_deleted;
       if total_deleted > WAL_CHECKPOINT_THRESHOLD {
           tracing::info!(total_deleted, "issuing WAL checkpoint after large prune");
           if let Err(e) = queries::wal_checkpoint(pool).await {
               tracing::error!(error = %e, "WAL checkpoint failed");
           }
       }

       tracing::info!(
           logs_deleted = total_logs_deleted,
           runs_deleted = total_runs_deleted,
           "retention prune cycle completed"
       );
   }
   ```
