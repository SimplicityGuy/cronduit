For `uat-labels-merge` and `uat-labels-reserved-namespace-error` (Scenario 4) — author NEW recipes that COMPOSE the existing `just check-config PATH` recipe (justfile L877) rather than re-inlining `cargo run -- check`. The runbook step is `just uat-labels-merge` (satisfies `feedback_uat_use_just_commands.md`); the recipe body composes `just check-config` (also a just recipe, also satisfies the rule).

Rationale for composing `check-config` inside the recipe body (resolves checker B2 part 2): `just check-config <path>` already wraps `cargo run --quiet -- check {{PATH}}`. Re-inlining the same `cargo run` invocation inside `uat-labels-*` would (a) duplicate the binary-invocation primitive in two places, (b) drift if `check-config` ever changes (e.g., adds new flags), and (c) leave raw `cargo run` strings in the justfile that the project memory rule is trying to discourage. Composing `just check-config` keeps the abstraction layer consistent.

```just
# UAT Scenario 4a: Custom Docker labels merge precedence — defaults + per-job merge.
# Composes `just check-config` (L877) rather than re-inlining `cargo run -- check`.
[group('uat')]
[doc('UAT Scenario 4a — labels merge precedence (defaults + per-job override) via check-config')]
uat-labels-merge:
    @mkdir -p .tmp
    @cat > .tmp/uat-labels-merge.toml <<'EOF'
    [defaults]
    labels = { "com.example.env" = "prod", "com.example.owner" = "platform" }

    [[jobs]]
    name = "label-merge-test"
    type = "docker"
    schedule = "0 0 * * *"
    image = "alpine:latest"
    command = "echo merged"
    labels = { "com.example.owner" = "data-team", "com.example.team" = "infra" }
    EOF
    just check-config .tmp/uat-labels-merge.toml
    @echo ""
    @echo "Maintainer eyeball: config parses without error. The per-job 'com.example.owner = data-team' MUST override the defaults 'platform' (per-job-wins on collision). New 'com.example.team = infra' MUST merge in."
    @read -p "Did the merge match the documented precedence (per-job-wins)? (y/n) " ok && [ "$$ok" = "y" ]

# UAT Scenario 4b: cronduit.* reserved-namespace error at config-load.
# Composes `just check-config` and asserts the validator REJECTS the config.
[group('uat')]
[doc('UAT Scenario 4b — cronduit.* reserved-namespace validator rejects operator-supplied label')]
uat-labels-reserved-namespace-error:
    @mkdir -p .tmp
    @cat > .tmp/uat-labels-reserved.toml <<'EOF'
    [[jobs]]
    name = "reserved-ns-test"
    type = "docker"
    schedule = "0 0 * * *"
    image = "alpine:latest"
    command = "echo reserved"
    labels = { "cronduit.job-name" = "operator-supplied" }
    EOF
    @echo "==> just check-config MUST FAIL with a reserved-namespace error"
    @! just check-config .tmp/uat-labels-reserved.toml 2>&1 | tee .tmp/uat-labels-reserved.log
    @grep -q "cronduit\." .tmp/uat-labels-reserved.log || (echo "FAIL: expected 'cronduit.*' reserved-namespace error message"; exit 1)
    @echo "PASS — reserved-namespace error surfaced at config-load."
```
