// Phase 22 TAG-01: build sorted-canonical JSON form for jobs.tags column.
// Sorted before serialization makes the column value stable across
// re-uploads of the same TOML and makes the substring-collision pass
// deterministic (RESEARCH § Section 1 + Pitfall 1).
let tags_json = {
    let mut sorted: Vec<String> = job.tags.iter()
        .map(|t| t.trim().to_lowercase())
        .filter(|t| !t.is_empty())
