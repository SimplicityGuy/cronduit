       ```rust
       /// POST /api/jobs/bulk-toggle ‚Äî set enabled_override for multiple jobs (ERG-01, ERG-02).
       ///
       /// Extractor order (axum 0.8 enforces body-consuming last):
       ///   1. State       (FromRequestParts)
       ///   2. CookieJar   (FromRequestParts ‚Äî axum_extra)
       ///   3. ExtraForm   (FromRequest ‚Äî consumes body; MUST be last)
       #[axum::debug_handler]
       pub async fn bulk_toggle(
           State(state): State<AppState>,
           cookies: CookieJar,
           ExtraForm(form): ExtraForm<BulkToggleForm>,
       ) -> impl IntoResponse {
           // 1. CSRF ‚Äî BEFORE any DB work (Landmine ¬ß5)
           let cookie_token = cookies
               .get(csrf::CSRF_COOKIE_NAME)
               .map(|c| c.value().to_string())
               .unwrap_or_default();
           if !csrf::validate_csrf(&cookie_token, &form.csrf_token) {
               return error_bulk_toast(
                   StatusCode::FORBIDDEN,
                   "Session expired ‚Äî refresh and try again.",
               );
           }

           // 2. Validate action (D-11)
           let new_override: Option<i64> = match form.action.as_str() {
               "disable" => Some(0),
               "enable" => None,
               _ => {
                   return error_bulk_toast(
                       StatusCode::BAD_REQUEST,
                       "Invalid bulk action.",
                   );
               }
           };

           // 3. Reject empty job_ids (UI-SPEC resolution; Landmine ¬ß9 + #[serde(default)])
           if form.job_ids.is_empty() {
               return error_bulk_toast(
                   StatusCode::BAD_REQUEST,
                   "No jobs selected.",
               );
           }

           // 4. Dedupe (D-12a)
           let ids: Vec<i64> = form
               .job_ids
               .into_iter()
               .collect::<BTreeSet<_>>()
               .into_iter()
               .collect();

           // 5. Count currently-running among selected via active_runs (Research Open Q 2)
           let running_count = {
               let runs = state.active_runs.read().await;
               ids.iter().filter(|id| runs.contains_key(id)).count() as u64
           };

           // 6. DB UPDATE (awaited)
           let updated = match queries::bulk_set_override(&state.pool, &ids, new_override).await {
               Ok(n) => n,
               Err(err) => {
                   tracing::error!(target: "cronduit.web", error = %err, "bulk_toggle: UPDATE failed");
                   return error_bulk_toast(
                       StatusCode::INTERNAL_SERVER_ERROR,
                       "Database error.",
                   );
               }
           };
           let not_found = (ids.len() as u64).saturating_sub(updated);

           // 7. Fire Reload AFTER DB commit (Landmine ¬ß6); do not await the result
           let (resp_tx, _resp_rx) = tokio::sync::oneshot::channel();
           if state
               .cmd_tx
               .send(SchedulerCmd::Reload { response_tx: resp_tx })
               .await
               .is_err()
           {
               return error_bulk_toast(
                   StatusCode::SERVICE_UNAVAILABLE,
                   "Scheduler is shutting down ‚Äî try again shortly.",
               );
           }
           drop(_resp_rx); // explicit: we don't await the reload result

           // 8. Compose toast per UI-SPEC Copywriting Contract
           let message = build_bulk_toast_message(updated, new_override, running_count, not_found);
           let event = HxEvent::new_with_data(
               "showToast",
               json!({"message": message, "level": "info"}),
           )
           .expect("toast event serialization");
           (HxResponseTrigger::normal([event]), StatusCode::OK).into_response()
       }

       /// Helper: build an error-toast response (sticky, level=error, duration=0).
       fn error_bulk_toast(status: StatusCode, message: &str) -> axum::response::Response {
           let event = HxEvent::new_with_data(
               "showToast",
               json!({"message": message, "level": "error", "duration": 0}),
           )
           .expect("toast event serialization");
           (HxResponseTrigger::normal([event]), status).into_response()
       }

       /// Helper: compose the verbose bulk-toggle success toast per UI-SPEC Copywriting.
       /// `updated`        ‚Äî rows actually UPDATEd (DB rows_affected)
       /// `new_override`   ‚Äî Some(0) for disable, None for enable, Some(1) is unused in UI
       /// `running_count`  ‚Äî jobs among the selection currently running (from active_runs)
       /// `not_found`      ‚Äî ids submitted that had no matching row
       fn build_bulk_toast_message(
           updated: u64,
           new_override: Option<i64>,
           running_count: u64,
           not_found: u64,
       ) -> String {
           let noun_updated = if updated == 1 { "job" } else { "jobs" };
           let mut msg = match new_override {
               Some(0) => format!("{} {} disabled.", updated, noun_updated),
               None => format!("{} {}: override cleared.", updated, noun_updated),
               Some(_) => format!("{} {}: override set.", updated, noun_updated), // reserved
           };
           // Conditional second sentence: running-job note applies only on disable
           if matches!(new_override, Some(0)) && running_count > 0 {
               let running_noun = if running_count == 1 { "job" } else { "jobs" };
               let running_verb = if running_count == 1 { "will" } else { "will" }; // both "will"
               msg.push_str(&format!(
                   " {} currently-running {} {} complete naturally.",
                   running_count, running_noun, running_verb
               ));
           }
           // Not-found suffix (D-12)
           if not_found > 0 {
               msg.push_str(&format!(" ({} not found)", not_found));
           }
           msg
       }
       ```
