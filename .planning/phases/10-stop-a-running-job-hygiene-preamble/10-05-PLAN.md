Some(cmd::SchedulerCmd::Stop { run_id, response_tx }) => {
    // Gap 4 / Option C: the map IS the race token.
    // If the executor called active_runs.write().await.remove(&run_id) at
    // run.rs:276 before this arm fires, lookup returns None and the race-case
    // branch replies AlreadyFinalized. No DB read, no extra coordination.
    //
    // Lock scope invariant: acquire read lock, clone out the control, release
    // the lock. The control.stop() call itself is cheap (atomic store +
    // CancellationToken.cancel()) but we release the lock first to keep the
    // invariant "no locks held across state changes" uniform with other arms.
    let maybe_control = {
        let active = self.active_runs.read().await;
        active.get(&run_id).map(|entry| entry.control.clone())
    };
    let result = match maybe_control {
        Some(control) => {
            control.stop(crate::scheduler::control::StopReason::Operator);
            tracing::info!(
                target: "cronduit.scheduler",
                run_id,
                "stop requested via command channel"
            );
            cmd::StopResult::Stopped
        }
        None => {
            tracing::debug!(
                target: "cronduit.scheduler",
                run_id,
                "Stop arrived after run finalized (race case)"
            );
            cmd::StopResult::AlreadyFinalized
        }
    };
    let _ = response_tx.send(result);
}

//!
//! Phase-gate blocker per D-15. The Stop feature does NOT ship unless this
//! test is green. Runs 1000 iterations under tokio::time::pause to get

#[tokio::test(flavor = "current_thread", start_paused = true)]
async fn stop_race_thousand_iterations() {
    for iter in 0..1000 {
        let pool = setup_pool().await;
        let job_id = 1_i64;
        let run_id = seed_running_run(&pool, job_id).await;

        let active_runs: Arc<RwLock<HashMap<i64, RunEntry>>> =
            Arc::new(RwLock::new(HashMap::new()));

        let (broadcast_tx, _rx) = broadcast::channel(256);
        let cancel = CancellationToken::new();
        let control = RunControl::new(cancel.clone());

        active_runs.write().await.insert(
            run_id,
            RunEntry {
                broadcast_tx: broadcast_tx.clone(),
                control: control.clone(),
                job_name: "race-test".to_string(),
            },
        );

        // Mock executor: finalizes to "success" at T+1ms OR observes cancel
        // and finalizes to "stopped". Whichever fires first wins; the other
        // branch must be a no-op (the .remove() + drop sequence on the
        // winning path ensures single-writer semantics per Invariant 3).
        let pool_clone = pool.clone();
        let active_runs_clone = active_runs.clone();
        let exec = tokio::spawn({
            let control = control.clone();
            async move {
                tokio::select! {
                    _ = tokio::time::sleep(Duration::from_millis(1)) => {
                        // Natural completion → finalize as success.
                        sqlx::query("UPDATE job_runs SET status = 'success', end_time = ?1 WHERE id = ?2 AND status = 'running'")
                            .bind(Utc::now().to_rfc3339())
                            .bind(run_id)
                            .execute(pool_clone.writer().as_sqlite().unwrap())
                            .await
                            .unwrap();
                        active_runs_clone.write().await.remove(&run_id);
                        drop(broadcast_tx);
                    }
                    _ = control.cancel.cancelled() => {
                        let status_str = match control.reason() {
                            StopReason::Operator => "stopped",
                            StopReason::Shutdown => "cancelled",
                        };
                        sqlx::query("UPDATE job_runs SET status = ?1, end_time = ?2 WHERE id = ?3 AND status = 'running'")
                            .bind(status_str)
                            .bind(Utc::now().to_rfc3339())
                            .bind(run_id)
                            .execute(pool_clone.writer().as_sqlite().unwrap())
                            .await
                            .unwrap();
                        active_runs_clone.write().await.remove(&run_id);
                        drop(broadcast_tx);
                    }
                }
            }
        });

        // Advance virtual time to T+999μs (1μs before natural exit).
        tokio::time::advance(Duration::from_micros(999)).await;

        // Fire the Stop.
        control.stop(StopReason::Operator);

        // Drain virtual time so the executor resolves one way or the other.
        tokio::time::advance(Duration::from_millis(2)).await;
        let _ = exec.await;

        // Assert the final row is ONE of {success, stopped} — never corrupted,
        // never "running", never anything else. The WHERE status = 'running'
        // guard in the UPDATE queries ensures single-writer semantics.
        let final_s = final_status(&pool, run_id).await;
        assert!(
            final_s == "success" || final_s == "stopped",
            "iteration {iter}: unexpected final status {final_s}"
        );

        pool.close().await;
    }
}

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Scheduler command channel | Web handlers (untrusted in the sense of "outside the scheduler event loop") enqueue `SchedulerCmd::Stop` via mpsc.
