   ```rust
   #[derive(Clone)]
   pub struct LogSender {
       state: Arc<Mutex<SharedState>>,
       notify: Arc<Notify>,
   }
   impl LogSender {
       pub fn send(&self, line: LogLine) {
           let mut state = self.state.lock().unwrap();
           if state.closed { return; } // receiver dropped
           if state.buf.len() >= state.capacity {
               state.buf.pop_front(); // HEAD-DROP: remove oldest (D-10)
               state.dropped_count += 1;
           }
           state.buf.push_back(line);
           drop(state); // release lock before notify
           self.notify.notify_one();
       }
       pub fn dropped_count(&self) -> usize {
           self.state.lock().unwrap().dropped_count
       }
       pub fn close(&self) {
           let mut state = self.state.lock().unwrap();
           state.closed = true;
           drop(state);
           self.notify.notify_one();
       }
   }
   ```

   e. Define `LogReceiver`:
   ```rust
   pub struct LogReceiver {
       state: Arc<Mutex<SharedState>>,
       notify: Arc<Notify>,
   }
   impl LogReceiver {
       /// Drain up to `max` lines from the buffer. If buffer was overflowed,
       /// prepend a "[truncated N lines]" marker as the first line of the batch.
       pub fn drain_batch(&self, max: usize) -> Vec<LogLine> {
           let mut state = self.state.lock().unwrap();
           let mut batch = Vec::with_capacity(max.min(state.buf.len()));

           // If lines were dropped, insert truncation marker FIRST
           if state.dropped_count > 0 {
               batch.push(LogLine {
                   stream: "system".to_string(),
                   ts: chrono::Utc::now().to_rfc3339(),
                   line: format!("[truncated {} lines]", state.dropped_count),
               });
               state.dropped_count = 0;
           }

           let drain_count = max.saturating_sub(batch.len()).min(state.buf.len());
           for _ in 0..drain_count {
               if let Some(line) = state.buf.pop_front() {
                   batch.push(line);
               }
           }
           batch
       }

       /// Wait for new lines to be available, then drain.
       pub async fn drain_batch_async(&self, max: usize) -> Vec<LogLine> {
           loop {
               let batch = self.drain_batch(max);
               if !batch.is_empty() { return batch; }
               let is_closed = self.state.lock().unwrap().closed;
               if is_closed { return vec![]; }
               self.notify.notified().await;
           }
       }

       pub fn is_empty_and_closed(&self) -> bool {
           let state = self.state.lock().unwrap();
           state.closed && state.buf.is_empty() && state.dropped_count == 0
       }
   }
   impl Drop for LogReceiver {
       fn drop(&mut self) {
           self.state.lock().unwrap().closed = true;
       }
   }
   ```

   ```rust
   pub async fn read_lines_to_channel<R: AsyncRead + Unpin>(
       reader: R,
       stream_name: &str,
       sender: LogSender,
   ) {
       let buf_reader = BufReader::new(reader);
       let mut lines = buf_reader.lines();
       while let Ok(Some(line)) = lines.next_line().await {
           sender.send(make_log_line(stream_name, line));

   }
   ```

   c. `pub async fn execute_command(command_str: &str, timeout: Duration, cancel: CancellationToken, sender: LogSender) -> ExecResult`:
   - Parse command via `shell_words::split(command_str)`. On parse error, return `ExecResult { exit_code: None, status: RunStatus::Error, error_message: Some(format!("failed to parse command: {}", e)) }`.
   - Build `tokio::process::Command::new(&argv[0]).args(&argv[1..])`.
   - Set `.stdout(Stdio::piped()).stderr(Stdio::piped())`.
   - Call `.process_group(0)` to put the child in its own process group (for clean kill on timeout).
   - Spawn the child. On spawn error, return Error status.
   - Take stdout and stderr handles, spawn two `read_lines_to_channel` tasks.
   - `tokio::select!` over: (a) `child.wait()`, (b) `tokio::time::sleep(timeout)`, (c) `cancel.cancelled()`.

   - On timeout: kill process group via `libc::kill(-(child.id().unwrap() as i32), libc::SIGKILL)`, wait for reader tasks, close sender, return Timeout.
   - On cancel: same kill + close, return Shutdown.
   - In all cases, call `sender.close()` to signal the log writer that no more lines are coming.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Operator config -> command executor | Command strings from TOML config are tokenized via shell-words, not passed to a shell. But the operator controls the binary, so this is a self-trust boundary. |
| Script body -> filesystem | Script bodies are written to tempfiles; path is system-controlled (not user-controlled). |
| Child process stdout/stderr -> log pipeline | Untrusted process output enters the log pipeline; bounded channel prevents resource exhaustion. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-04 | Tampering | execute_command | accept | Operator writes their own commands; shell-words prevents shell injection from config values. This is self-trust: the operator already has Docker socket access. |
| T-02-05 | Denial of Service | log_pipeline | mitigate | Bounded channel (256 lines) with head-drop prevents OOM from chatty processes. Line truncation at 16 KB prevents single-line memory bombs. |
| T-02-06 | Denial of Service | execute_command/script | mitigate | Process group kill on timeout ensures child processes and their children are killed, preventing zombie accumulation. |
| T-02-07 | Elevation of Privilege | execute_script tempfile | accept | Tempfile is in system temp dir with random name. If /tmp is noexec, script execution fails with a clear error. Operator controls script content (self-trust). |
| T-02-08 | Information Disclosure | log_pipeline | accept | Log content from child processes is stored in job_logs DB table. No additional exposure beyond what the operator's own commands produce. |
</threat_model>
