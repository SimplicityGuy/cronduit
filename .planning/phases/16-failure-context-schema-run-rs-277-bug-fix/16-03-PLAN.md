//! Phase 16 / FOUND-14: assert that v1.2 docker runs persist the REAL Docker
//! container ID into job_runs.container_id (not the sha256: image digest that
//! v1.0/v1.1 silently stored due to the run.rs:301 bug). Also asserts the
//! parallel image_digest column is populated correctly for docker jobs and
//! left NULL for command/script jobs.
//!

//! T-V12-FCTX-08: command run leaves image_digest = NULL.

use bollard::Docker;
use cronduit::db::DbPool;
use cronduit::db::queries::PoolRef;
use sqlx::Row;

async fn docker_client() -> Docker {
    Docker::connect_with_local_defaults()
        .expect("Docker daemon must be running for integration tests")
}

async fn setup_test_db() -> DbPool {
    let pool = DbPool::connect("sqlite::memory:").await.unwrap();
    pool.migrate().await.unwrap();
    pool
}

#[tokio::test]
#[ignore]
async fn docker_run_writes_real_container_id_not_digest() {
    let _docker = docker_client().await;
    let pool = setup_test_db().await;
    // 1. Insert a job with type = "docker" pointing at alpine:latest with `echo hello`.
    // 2. Drive a run via run_job (or, if run_job's full setup is too heavy, drive
    //    execute_docker + manual insert_running_run + finalize_run mirroring the
    //    production sequence at run.rs:79-348). The KEY is that finalize_run runs.

    // 4. Assert container_id is Some(_) AND !starts_with("sha256:").
    //    The historical bug stored sha256:... in this column; v1.2 must not.
    let p = match pool.reader() {
        PoolRef::Sqlite(p) => p,
        _ => panic!("sqlite-only test"),
    };
    let row = sqlx::query("SELECT container_id, image_digest FROM job_runs WHERE id = ?1")
        .bind(/* run_id from above */ 1i64)
        .fetch_one(p).await.unwrap();
    let container_id: Option<String> = row.get("container_id");
    let cid = container_id.expect("container_id must be Some after a successful docker run");
    assert!(!cid.starts_with("sha256:"),
            "Phase 16 FOUND-14: container_id must be the real Docker container ID, not the sha256 digest. Got: {cid}");
}
