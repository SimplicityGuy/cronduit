    fn check_schedule(job: &JobConfig, path: &Path, errors: &mut Vec<ConfigError>) {
        if let Err(e) = Cron::new(&job.schedule).parse() {
            errors.push(ConfigError {
                file: path.into(),
                line: 0,
                col: 0,
                message: format!(
                    "[[jobs]] `{}`: invalid cron expression `{}`: {}",
                    job.name, job.schedule, e
                ),
            });
        }
    }
