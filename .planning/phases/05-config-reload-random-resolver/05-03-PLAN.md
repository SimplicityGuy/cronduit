In the `run()` method's `tokio::select!` block, there is currently a branch:
```rust
cmd = self.cmd_rx.recv() => {
    match cmd {
        Some(SchedulerCmd::RunNow { job_id }) => { ... }
        None => break,
    }
}
```

Replace it with:
```rust
cmd = self.cmd_rx.recv() => {
    match cmd {
        Some(SchedulerCmd::RunNow { job_id }) => {
            // ... existing RunNow code unchanged ...
        }
        Some(SchedulerCmd::Reload { response_tx }) => {
            let (result, new_heap_opt) = reload::do_reload(
                &self.pool,
                &self.config_path,
                &mut self.jobs,
                self.tz,
            ).await;

            if let Some(new_heap) = new_heap_opt {
                heap = new_heap;
                // Update jobs_vec for clock-jump detection
                jobs_vec = self.jobs.values().cloned().collect();
            }

            let _ = response_tx.send(result);
        }
        Some(SchedulerCmd::Reroll { job_id, response_tx }) => {
            let (result, new_heap_opt) = reload::do_reroll(
                &self.pool,
                job_id,
                &mut self.jobs,
                self.tz,
            ).await;

            if let Some(new_heap) = new_heap_opt {
                heap = new_heap;
                jobs_vec = self.jobs.values().cloned().collect();
            }

            let _ = response_tx.send(result);
        }
        None => break,
    }
}
```

```rust
/// POST /api/reload -- trigger config reload (RELOAD-02, D-01, D-02, D-03).
pub async fn reload(
    State(state): State<AppState>,
    cookies: CookieJar,
    axum::Form(form): axum::Form<CsrfForm>,
) -> impl IntoResponse {
    // 1. Validate CSRF (T-05-08)
    let cookie_token = cookies
        .get(csrf::CSRF_COOKIE_NAME)
        .map(|c| c.value().to_string())
        .unwrap_or_default();

    if !csrf::validate_csrf(&cookie_token, &form.csrf_token) {
        return (StatusCode::FORBIDDEN, "CSRF token mismatch").into_response();
    }

    // 2. Send Reload command (D-09: through same SchedulerCmd channel)
    let (resp_tx, resp_rx) = tokio::sync::oneshot::channel();
    match state.cmd_tx.send(SchedulerCmd::Reload { response_tx: resp_tx }).await {
        Ok(()) => {
            match resp_rx.await {
                Ok(result) => {
                    // D-03: JSON diff summary
                    let status_str = match result.status {
                        crate::scheduler::cmd::ReloadStatus::Ok => "ok",
                        crate::scheduler::cmd::ReloadStatus::Error => "error",
                    };
                    let json_body = json!({
                        "status": status_str,
                        "added": result.added,
                        "updated": result.updated,
                        "disabled": result.disabled,
                        "unchanged": result.unchanged,
                        "message": result.error_message,
                    });

                    // D-01/D-02: Toast with diff summary or error
                    let (toast_msg, toast_level, toast_duration) = match result.status {
                        crate::scheduler::cmd::ReloadStatus::Ok => {
                            let msg = format!(
                                "Config reloaded: {} added, {} updated, {} disabled",
                                result.added, result.updated, result.disabled
                            );
                            (msg, "success", 5000u32)
                        }
                        crate::scheduler::cmd::ReloadStatus::Error => {
                            let msg = format!(
                                "Reload failed: {}",
                                result.error_message.as_deref().unwrap_or("unknown error")
                            );
                            (msg, "error", 0u32)  // 0 = persist until dismissed (D-02)
                        }
                    };

                    let event = HxEvent::new_with_data(
                        "showToast",
                        json!({"message": toast_msg, "level": toast_level, "duration": toast_duration}),
                    ).expect("toast event serialization");

                    (
                        HxResponseTrigger::normal([event]),
                        axum::Json(json_body),
                    ).into_response()
                }
                Err(_) => (StatusCode::SERVICE_UNAVAILABLE, "Scheduler shutting down").into_response(),
            }
        }
        Err(_) => (StatusCode::SERVICE_UNAVAILABLE, "Scheduler shutting down").into_response(),
    }
}

/// POST /api/jobs/{id}/reroll -- re-roll @random schedule (D-06).
pub async fn reroll(
    State(state): State<AppState>,
    Path(job_id): Path<i64>,
    cookies: CookieJar,
    axum::Form(form): axum::Form<CsrfForm>,
) -> impl IntoResponse {
    // 1. Validate CSRF
    let cookie_token = cookies
        .get(csrf::CSRF_COOKIE_NAME)
        .map(|c| c.value().to_string())
        .unwrap_or_default();

    if !csrf::validate_csrf(&cookie_token, &form.csrf_token) {
        return (StatusCode::FORBIDDEN, "CSRF token mismatch").into_response();
    }

    // 2. Verify job exists
    let job = match queries::get_job_by_id(&state.pool, job_id).await {
        Ok(Some(job)) => job,
        Ok(None) => return (StatusCode::NOT_FOUND, "Job not found").into_response(),
        Err(_) => return (StatusCode::INTERNAL_SERVER_ERROR, "Database error").into_response(),
    };

    // 3. Send Reroll command
    let (resp_tx, resp_rx) = tokio::sync::oneshot::channel();
    match state.cmd_tx.send(SchedulerCmd::Reroll { job_id, response_tx: resp_tx }).await {
        Ok(()) => {
            match resp_rx.await {
                Ok(result) => {
                    let (toast_msg, toast_level, toast_duration) = match result.status {
                        crate::scheduler::cmd::ReloadStatus::Ok => {
                            (format!("Schedule re-rolled for {}", job.name), "success", 5000u32)
                        }
                        crate::scheduler::cmd::ReloadStatus::Error => {
                            let msg = result.error_message.as_deref()
                                .unwrap_or("This job has no @random schedule");
                            (msg.to_string(), "error", 0u32)
                        }
                    };

                    let event = HxEvent::new_with_data(
                        "showToast",
                        json!({"message": toast_msg, "level": toast_level, "duration": toast_duration}),
                    ).expect("toast event serialization");

                    // HX-Refresh: true to reload the job detail page with new resolved schedule
                    let mut headers = axum::http::HeaderMap::new();
                    if result.status == crate::scheduler::cmd::ReloadStatus::Ok {
                        headers.insert("HX-Refresh", "true".parse().unwrap());
                    }

                    (
                        HxResponseTrigger::normal([event]),
                        headers,
                        StatusCode::OK,
                    ).into_response()
                }
                Err(_) => (StatusCode::SERVICE_UNAVAILABLE, "Scheduler shutting down").into_response(),
            }
        }
        Err(_) => (StatusCode::SERVICE_UNAVAILABLE, "Scheduler shutting down").into_response(),
    }
}
```
