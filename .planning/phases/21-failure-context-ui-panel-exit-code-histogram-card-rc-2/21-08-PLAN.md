```rust
async fn seed_runs_with_status_and_exit(
    pool: &DbPool,
    job_id: i64,
    count: usize,
    status: &str,
    exit_code: Option<i32>,
) {
    let p = match pool.writer() {
        PoolRef::Sqlite(p) => p,
        _ => panic!("sqlite-only test"),
    };
    for i in 0..count {
        let start_time = format!("2026-04-27T{:02}:{:02}:00Z", (i / 60) % 24, i % 60);
        let end_time   = format!("2026-04-27T{:02}:{:02}:30Z", (i / 60) % 24, i % 60);
        sqlx::query(
            "INSERT INTO job_runs (job_id, status, trigger, start_time, end_time, duration_ms, exit_code, job_run_number, config_hash) \
             VALUES (?1, ?2, 'manual', ?3, ?4, 30000, ?5, ?6, 'seed-hash')",
        )
        .bind(job_id)
        .bind(status)
        .bind(&start_time)
        .bind(&end_time)
        .bind(exit_code)
        .bind((i as i64) + 1)
        .execute(p)
        .await
        .expect("seed run");
    }
}
```
