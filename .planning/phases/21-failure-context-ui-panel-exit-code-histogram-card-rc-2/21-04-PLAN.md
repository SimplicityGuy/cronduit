```rust
async fn build_fctx_view(
    run: &RunDetailView,
    ctx: FailureContext,
    pool: &DbPool,
) -> FctxView {
    // 1. summary_meta — UI-SPEC § Copywriting:
    //    streak > 1: "{N} consecutive failures"
    //    streak == 1: "1 failure (no streak)"
    let summary_meta = if ctx.consecutive_failures > 1 {
        format!("{} consecutive failures", ctx.consecutive_failures)
    } else {
        "1 failure (no streak)".to_string()
    };

    // 2. is_docker_job — fetch DbJob.job_type via the existing job-lookup helper.
    //    is_docker_job = job_type == "docker"
    let job = queries::get_job_by_id(pool, run.job_id).await.ok();  // soft-fail to non-docker on lookup error
    let is_docker_job = job.as_ref().map(|j| j.job_type == "docker").unwrap_or(false);

    // 3. last_success_run_url — only when last_success_run_id is Some.
    let last_success_run_url = ctx.last_success_run_id
        .map(|id| format!("/jobs/{}/runs/{}", run.job_id, id));

    // 4. time_deltas_value — D-13 branching:
    //    last_success_run_id.is_some() → "First failure: {rel} ago • {N} consecutive failures • [view last successful run]"
    //    last_success_run_id.is_none()  → "First failure: {rel} ago • {N} consecutive failures • No prior successful run"
    //    The `[view last successful run]` literal is the LINK TEXT — the template wraps it in <a> using last_success_run_url.
    //    Compute ts_relative from run.start_time RFC3339 vs Utc::now().
    let ts_relative = format_relative_time(&run.start_time);  // implement or reuse existing
    let time_deltas_value = if ctx.last_success_run_id.is_some() {
        format!("First failure: {} ago • {} consecutive failures", ts_relative, ctx.consecutive_failures)
    } else {
        format!("First failure: {} ago • {} consecutive failures • No prior successful run", ts_relative, ctx.consecutive_failures)
    };
    // Note: when last_success_run_id is Some, the template appends the [view last successful run] link via `last_success_run_url`.
    // The Rust-side string omits the link text; the template renders the link conditionally.

    // 5. image_digest_value — D-13: hide row when never-succeeded.
    let image_digest_value = if !is_docker_job {
        None  // FCTX-03 hide-on-non-docker (gated by is_docker_job in template too)
    } else if ctx.last_success_image_digest.is_none() {
        None  // D-13 never-succeeded → hide
    } else {
        let last = ctx.last_success_image_digest.as_deref().unwrap();
        let cur = run.image_digest.as_deref();
        match cur {
            None => Some("unchanged".to_string()),  // current run has no digest yet — treat as unchanged (defensive)
            Some(c) if c == last => Some("unchanged".to_string()),
            Some(c) => {
                let old_short = truncate_hex(last, 12);
                let new_short = truncate_hex(c, 12);
                Some(format!("{}… → {}…", old_short, new_short))
            }
        }
    };

    // 6. config_changed_value — D-14: literal compare; D-13: hide when never-succeeded.
    let config_changed_value = if ctx.last_success_config_hash.is_none() {
        None  // never-succeeded → hide row
    } else {
        let changed = run.config_hash.as_deref() != ctx.last_success_config_hash.as_deref();
        Some(if changed {
            "Config changed since last success: Yes".to_string()
        } else {
            "Config changed since last success: No".to_string()
        })
    };

    // 7. has_duration_samples + duration_value — UI-SPEC FCTX-05: N >= 5 successful samples.
    const FCTX_MIN_DURATION_SAMPLES: usize = 5;
    const FCTX_DURATION_SAMPLE_LIMIT: i64 = 100;
    let durations = queries::get_recent_successful_durations(pool, run.job_id, FCTX_DURATION_SAMPLE_LIMIT)
        .await
        .unwrap_or_default();
    let has_duration_samples = durations.len() >= FCTX_MIN_DURATION_SAMPLES;
    let duration_value = if has_duration_samples {
        let p50 = stats::percentile(&durations, 0.5)
            .expect("non-empty when has_duration_samples is true");
        let cur_ms = run.duration_ms.unwrap_or(0);
        let factor = if p50 > 0 { cur_ms as f64 / p50 as f64 } else { 1.0 };
        let direction = if factor >= 1.0 { "longer" } else { "shorter" };
        let factor_display = if factor >= 1.0 { factor } else { 1.0 / factor };
        Some(format!(
            "{}; typical p50 is {} ({:.1}× {} than usual)",
            format_duration_ms_floor_seconds(Some(cur_ms)),
            format_duration_ms_floor_seconds(Some(p50 as i64)),
            factor_display,
            direction,
        ))
    } else {
        None
    };

    // 8. fire_skew_value — D-04: hide row when scheduled_for IS NULL.
    let fire_skew_value = match (run.scheduled_for.as_deref(), run.start_time.as_str()) {
        (Some(sched), start) => {
            let sched_dt = chrono::DateTime::parse_from_rfc3339(sched).ok();
            let start_dt = chrono::DateTime::parse_from_rfc3339(start).ok();
            match (sched_dt, start_dt) {
                (Some(s), Some(t)) => {
                    let skew_ms = (t - s).num_milliseconds();
                    Some(format!(
                        "Scheduled: {} • Started: {} (+{} ms)",
                        s.format("%H:%M:%S"),
                        t.format("%H:%M:%S"),
                        skew_ms
                    ))
                }
                _ => None,  // unparseable timestamp → hide row defensively
            }
        }
        (None, _) => None,
    };

    FctxView {
        consecutive_failures: ctx.consecutive_failures,
        summary_meta,
        last_success_run_id: ctx.last_success_run_id,
        time_deltas_value,
        last_success_run_url,
        is_docker_job,
        image_digest_value,
        config_changed_value,
        has_duration_samples,
        duration_value,
        fire_skew_value,
    }
}
```
