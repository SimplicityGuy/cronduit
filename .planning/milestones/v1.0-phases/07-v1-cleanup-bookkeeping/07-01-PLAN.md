  #     # ports:
  #     #   - "8080:8080"
  #     expose:
  #       - "8080"
  #
  # With `expose:`, the container still listens on 8080 but the port is
  # NOT published on the host -- only other containers on the same Docker
  # network (e.g., your reverse proxy) can reach it.
  #
