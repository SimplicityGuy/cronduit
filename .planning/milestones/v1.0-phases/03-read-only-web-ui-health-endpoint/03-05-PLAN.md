1. **src/web/csrf.rs** — CSRF double-submit cookie implementation (per D-11, UI-15):

```rust
//! Double-submit CSRF token protection.
//!
//! D-11: Random token in HttpOnly; SameSite=Strict cookie + hidden form field.
//! Server validates they match. No server-side session store needed.

use rand::Rng;

/// Generate a random 32-byte CSRF token as a 64-char hex string.
pub fn generate_csrf_token() -> String {
    let token: [u8; 32] = rand::thread_rng().gen();
    hex::encode(token)
}

/// Validate that cookie token and form token match.
/// Uses constant-time comparison to prevent timing attacks.
pub fn validate_csrf(cookie_token: &str, form_token: &str) -> bool {
    if cookie_token.is_empty() || form_token.is_empty() {
        return false;
    }
    // Simple equality is acceptable for v1 — timing attacks on CSRF tokens
    // are theoretical when tokens are random hex. For defense-in-depth,
    // compare byte-by-byte to avoid short-circuit.
    let a = cookie_token.as_bytes();
    let b = form_token.as_bytes();
    if a.len() != b.len() {
        return false;
    }
    let mut result: u8 = 0;
    for (x, y) in a.iter().zip(b.iter()) {
        result |= x ^ y;
    }
    result == 0
}

/// Cookie name for the CSRF token.
pub const CSRF_COOKIE_NAME: &str = "cronduit_csrf";

/// Form field name for the CSRF token.
pub const CSRF_FIELD_NAME: &str = "csrf_token";

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn generate_token_is_64_hex_chars() {
        let token = generate_csrf_token();
        assert_eq!(token.len(), 64);
        assert!(token.chars().all(|c| c.is_ascii_hexdigit()));
    }

    #[test]
    fn matching_tokens_validate() {
        assert!(validate_csrf("abc123def456", "abc123def456"));
    }

    #[test]
    fn mismatched_tokens_reject() {
        assert!(!validate_csrf("abc123", "xyz789"));
    }

    #[test]
    fn empty_cookie_rejects() {
        assert!(!validate_csrf("", "abc123"));
    }

    #[test]
    fn empty_form_field_rejects() {
        assert!(!validate_csrf("abc123", ""));
    }
}
```

```rust
pub async fn run_now(
    State(state): State<AppState>,
    Path(job_id): Path<i64>,
    cookies: axum_extra::extract::CookieJar,
    axum::Form(form): axum::Form<RunNowForm>,
) -> impl IntoResponse {
    // 1. Validate CSRF token
    let cookie_token = cookies
        .get(csrf::CSRF_COOKIE_NAME)
        .map(|c| c.value().to_string())
        .unwrap_or_default();

    if !csrf::validate_csrf(&cookie_token, &form.csrf_token) {
        return (StatusCode::FORBIDDEN, "CSRF token mismatch").into_response();
    }

    // 2. Verify job exists
    let job = match queries::get_job_by_id(&state.pool, job_id).await {
        Ok(Some(job)) => job,
        Ok(None) => return (StatusCode::NOT_FOUND, "Job not found").into_response(),
        Err(_) => return (StatusCode::INTERNAL_SERVER_ERROR, "Database error").into_response(),
    };

    // 3. Send RunNow command through scheduler channel (D-08)
    match state.cmd_tx.send(SchedulerCmd::RunNow { job_id }).await {
        Ok(()) => {
            tracing::info!(
                target: "cronduit.web",
                job_id,
                job_name = %job.name,
                "Run Now requested via API"
            );

            // 4. Return 204 with HX-Trigger for toast (D-10)
            let trigger = HxResponseTrigger::normal(["showToast"]);
            // Include job name in a custom header for the toast message
            (
                StatusCode::NO_CONTENT,
                trigger,
                [("X-Toast-Message", format!("Run queued: {}", job.name))],
            ).into_response()
        }
        Err(_) => {
            (StatusCode::SERVICE_UNAVAILABLE, "Scheduler is shutting down").into_response()
        }
    }
}
```

3. **CSRF cookie middleware** — Add a tower layer or axum middleware that sets the CSRF cookie on every GET response if not already present. Add this as a function in `csrf.rs`:

```rust
/// Axum middleware that ensures the CSRF cookie is set on every response.
pub async fn ensure_csrf_cookie(
    cookies: axum_extra::extract::CookieJar,
    request: axum::extract::Request,
    next: axum::middleware::Next,
) -> impl IntoResponse {
    let has_cookie = cookies.get(CSRF_COOKIE_NAME).is_some();
    let mut response = next.run(request).await;

    if !has_cookie {
        let token = generate_csrf_token();
        let cookie = format!(
            "{}={}; HttpOnly; SameSite=Strict; Path=/",
            CSRF_COOKIE_NAME, token
        );
        response.headers_mut().insert(
            axum::http::header::SET_COOKIE,
            cookie.parse().unwrap(),
        );
    }

    response
}
```

Also add a helper for templates to read the token from cookies:
```rust
/// Extract the CSRF token from the cookie jar, or generate a new one.
pub fn get_token_from_cookies(cookies: &axum_extra::extract::CookieJar) -> String {
    cookies
        .get(CSRF_COOKIE_NAME)
        .map(|c| c.value().to_string())
```
