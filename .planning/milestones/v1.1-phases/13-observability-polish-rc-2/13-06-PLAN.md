    ```

    let pg_sql = sql.replace("?1", "$1").replace("j.enabled = 1", "j.enabled = true");
    let explain_sql = format!("EXPLAIN (FORMAT JSON) {pg_sql}");

    let row = match pool.reader() {
        cronduit::db::queries::PoolRef::Postgres(p) => {
            // REQUIRED: Fresh testcontainer statistics default to guessing
            // cardinality, which often picks Seq Scan even when an index exists.
            // ANALYZE forces the planner to consult real row counts.
            sqlx::query("ANALYZE job_runs").execute(p).await.expect("analyze");

            sqlx::query(&explain_sql).bind(window_start).fetch_one(p).await.expect("explain")
        }
        _ => panic!("expected pg pool"),
    };

    let plan_json: serde_json::Value = row.get(0);
    // Walk the plan tree looking for any node with "Node Type" matching Index Scan / Index Only Scan
    fn contains_index_scan(v: &serde_json::Value) -> bool {
        if let Some(node_type) = v.get("Node Type").and_then(|s| s.as_str()) {
            if node_type == "Index Scan" || node_type == "Index Only Scan" {
                return true;
            }
        }
        if let Some(plans) = v.get("Plans").and_then(|p| p.as_array()) {
            return plans.iter().any(contains_index_scan);
        }
        if let Some(arr) = v.as_array() {
            return arr.iter().any(contains_index_scan);
        }
        false
    }

    assert!(
        contains_index_scan(&plan_json),
        "expected Postgres EXPLAIN JSON to contain Index Scan on job_runs; got:\n{plan_json:#}"
    );
    ```

            VALUES (?, 1, 'success', 'scheduled', ?, ?, 60000, 0)";
