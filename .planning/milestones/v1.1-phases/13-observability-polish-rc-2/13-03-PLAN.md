        1..=19 => format!("{sample_count} of 20 successful runs required"),
        20..=99 => format!("last {sample_count} successful runs"),
        _ => "last 100 successful runs".to_string(),

    let duration_view = DurationView {

        p95_display,
        has_min_samples: has_min,
