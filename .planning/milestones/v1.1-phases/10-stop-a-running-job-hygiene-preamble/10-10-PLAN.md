use cronduit::db::DbPool;
use cronduit::scheduler::cmd::{SchedulerCmd, StopResult};
use tokio::sync::{mpsc, RwLock, oneshot};

    use_project_upsert_job(pool, "test-stop-cmd", "command", &format!(r#"{{"command":"{cmd_str}"}}"#)).await
}

async fn read_run_status(pool: &DbPool, run_id: i64) -> String {

#[tokio::test]
async fn stop_command_executor_yields_stopped_status() {
    let pool = DbPool::connect("sqlite::memory:").await.expect("pool");
    pool.migrate().await.expect("migrate");
    let job_id = seed_command_job(&pool, "sleep 30").await;

    let (cmd_tx, cmd_rx) = mpsc::channel::<SchedulerCmd>(16);
    let active_runs = Arc::new(RwLock::new(std::collections::HashMap::new()));
    let cancel = CancellationToken::new();

    // Start a real scheduler loop (abbreviated — use scheduler::spawn with
    // a single-job jobs_vec and the mpsc receiver).
    let scheduler_handle = start_test_scheduler_loop(
        pool.clone(), cancel.clone(), cmd_rx, active_runs.clone(), job_id,
    ).await;

    // Trigger the run.
    cmd_tx.send(SchedulerCmd::RunNow { job_id }).await.expect("run now");
    // Wait for the run to enter active_runs.
    tokio::time::sleep(Duration::from_millis(200)).await;

    // Find the running run_id.
    let run_id = {
        let map = active_runs.read().await;
        *map.keys().next().expect("at least one running run")
    };

    // Dispatch Stop via the scheduler channel.
    let (resp_tx, resp_rx) = oneshot::channel();
    cmd_tx.send(SchedulerCmd::Stop { run_id, response_tx: resp_tx }).await.expect("stop send");
    assert_eq!(resp_rx.await.expect("stop reply"), StopResult::Stopped);

    // Wait for the executor to finalize.
    tokio::time::sleep(Duration::from_millis(500)).await;

    // Assert DB row.
    let status = read_run_status(&pool, run_id).await;
    assert_eq!(status, "stopped", "T-V11-STOP-09: command executor must finalize stopped runs with status='stopped'");

    cancel.cancel();
    let _ = scheduler_handle.await;
}

#[tokio::test]
async fn stop_kills_shell_pipeline_grandchildren() {

    // (sleep + 2 cats) connected by pipes. The direct child is `sh`; the
    // grandchildren are `sleep`, `cat`, `cat`.

    // Send Stop through the scheduler channel. Wait for the executor to finalize.
    //
    // Assert that NONE of the captured grandchild pids still exist in /proc.
    // kill_on_drop(true) would leave sleep + cat processes alive because
    // kill_on_drop only targets the direct child, not its descendants.
    //
    // Implementation hint: a simpler version of this test uses a sentinel
    // file. Replace the pipeline with `sh -c 'sleep 300 && touch /tmp/sentinel-<uuid>'`
    // — after Stop + a 1s wait, assert the sentinel file does NOT exist.
    // This avoids /proc parsing.

async fn stop_kills_backgrounded_processes_in_script() {
    // Seed a script job with body:
    //   #!/bin/sh
    //   (sleep 300 && touch /tmp/bg-sentinel-<uuid>) &
    //   sleep 300
    //
    // The backgrounded subshell sleeps 5 minutes and then writes a sentinel.
    // The main script also sleeps 5 minutes.
    //
    // Start the run. Wait 200ms. Send Stop. Wait 1s.
    //
    // Assert the sentinel file does NOT exist. If kill_on_drop(true) were
    // adopted, the main sh would die but the backgrounded subshell would
    // survive because it's in a different process but the SAME process group
    // (since the parent `sh` used setpgid=0 via .process_group(0), the whole
    // group dies on `kill(-pgid)`).
}
