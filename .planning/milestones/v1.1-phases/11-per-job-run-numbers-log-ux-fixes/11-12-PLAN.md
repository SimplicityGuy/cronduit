#[tokio::test]
async fn data_max_id_rendered() {
    let (app, state) = tests_common::build_test_app().await;
    let job_id = seed_test_job(&state.pool, "maxid-job").await;
    let run_id = seed_running_run(&state.pool, job_id).await;
    let batch = make_test_batch(5);
    let ids = cronduit::db::queries::insert_log_batch(&state.pool, run_id, &batch).await.unwrap();
    let max = ids.last().unwrap();

    let response = app.oneshot(Request::builder()
        .method("GET").uri(format!("/jobs/{}/runs/{}", job_id, run_id))
        .body(Body::empty()).unwrap()).await.unwrap();
    let body = String::from_utf8_lossy(
        &axum::body::to_bytes(response.into_body(), 10 * 1024 * 1024).await.unwrap());

    assert!(body.contains(&format!("data-max-id=\"{}\"", max)),
        "data-max-id must equal {} (last inserted log id): {}", max,
        &body.chars().take(800).collect::<String>());
}
