```rust
use crate::db::queries;

impl DbPool {
    pub async fn migrate(&self) -> anyhow::Result<()> {
        // Check whether file 3 (NOT NULL) can apply immediately without constraint
        // violation. Safe when:
        //   - job_runs table does not yet exist (fresh install)
        //   - OR job_run_number column does not yet exist AND job_runs is empty
        //   - OR job_run_number column exists and has zero NULL values
        let file3_safe = self.file3_can_apply_now().await.unwrap_or(false);

        if file3_safe {
            // Single-pass: apply everything in one go. Orchestrator then
            // short-circuits via sentinel / zero-null-count.
            match self {
                DbPool::Sqlite(p) => sqlx::migrate!("./migrations/sqlite").run(p).await?,
                DbPool::Postgres(p) => sqlx::migrate!("./migrations/postgres").run(p).await?,
            }
            migrate_backfill::backfill_job_run_number(self).await?;
        } else {
            // Upgrade-in-place with NULL rows. Apply files 1+2 only, run
            // orchestrator to fill NULLs, then apply file 3.
            self.migrate_up_to_backfill_marker().await?;
            migrate_backfill::backfill_job_run_number(self).await?;
            match self {
                DbPool::Sqlite(p) => sqlx::migrate!("./migrations/sqlite").run(p).await?,
                DbPool::Postgres(p) => sqlx::migrate!("./migrations/postgres").run(p).await?,
            }
        }
        Ok(())
    }

    async fn file3_can_apply_now(&self) -> anyhow::Result<bool> {
        // Check whether job_runs table exists.
        let table_exists = match self {
            DbPool::Sqlite(p) => {
                let n: i64 = sqlx::query_scalar(
                    "SELECT COUNT(*) FROM sqlite_master \
                     WHERE type = 'table' AND name = 'job_runs'",
                )
                .fetch_one(p)
                .await
                .unwrap_or(0);
                n > 0
            }
            DbPool::Postgres(p) => {
                sqlx::query_scalar::<_, bool>(
                    "SELECT EXISTS ( \
                        SELECT 1 FROM information_schema.tables \
                        WHERE table_name = 'job_runs' \
                     )",
                )
                .fetch_one(p)
                .await
                .unwrap_or(false)
            }
        };
        if !table_exists {
            return Ok(true);
        }

        // Check whether job_run_number column exists. If not, table was created
        // before Phase 11. If it has any rows, they'll all be NULL after file 1
        // adds the column, making file 3 unsafe. If empty, safe.
        let column_exists = match self {
            DbPool::Sqlite(p) => {
                let n: i64 = sqlx::query_scalar(
                    "SELECT COUNT(*) FROM pragma_table_info('job_runs') \
                     WHERE name = 'job_run_number'",
                )
                .fetch_one(p)
                .await
                .unwrap_or(0);
                n > 0
            }
            DbPool::Postgres(p) => sqlx::query_scalar::<_, bool>(
                "SELECT EXISTS ( \
                    SELECT 1 FROM information_schema.columns \
                    WHERE table_name = 'job_runs' \
                      AND column_name = 'job_run_number' \
                 )",
            )
            .fetch_one(p)
            .await
            .unwrap_or(false),
        };
        if !column_exists {
            let row_count: i64 = match self {
                DbPool::Sqlite(p) => sqlx::query_scalar("SELECT COUNT(*) FROM job_runs")
                    .fetch_one(p)
                    .await
                    .unwrap_or(0),
                DbPool::Postgres(p) => sqlx::query_scalar("SELECT COUNT(*) FROM job_runs")
                    .fetch_one(p)
                    .await
                    .unwrap_or(0),
            };
            return Ok(row_count == 0);
        }

        // Column exists — count NULLs directly.
        let null_count = queries::count_job_runs_with_null_run_number(self).await?;
        Ok(null_count == 0)
    }

    /// Applies every migration with version <= 20260416_000002 (file 2 marker),
    /// so the orchestrator can fill NULLs before file 3's NOT NULL tightening.
    async fn migrate_up_to_backfill_marker(&self) -> anyhow::Result<()> {
        const FILE2_VERSION: i64 = 20260416_000002;
        match self {
            DbPool::Sqlite(p) => {
                let m: &sqlx::migrate::Migrator = &sqlx::migrate!("./migrations/sqlite");
                for migration in m.iter() {
                    if migration.version > FILE2_VERSION {
                        continue;
                    }
                    let applied: i64 = sqlx::query_scalar(
                        "SELECT COUNT(*) FROM _sqlx_migrations WHERE version = ?1",
                    )
                    .bind(migration.version)
                    .fetch_one(p)
                    .await
                    .unwrap_or(0);
                    if applied > 0 {
                        continue;
                    }
                    migration.apply(p).await?;
                }
            }
            DbPool::Postgres(p) => {
                let m: &sqlx::migrate::Migrator = &sqlx::migrate!("./migrations/postgres");
                for migration in m.iter() {
                    if migration.version > FILE2_VERSION {
                        continue;
                    }
                    let applied: i64 = sqlx::query_scalar(
                        "SELECT COUNT(*) FROM _sqlx_migrations WHERE version = $1",
                    )
                    .bind(migration.version)
                    .fetch_one(p)
                    .await
                    .unwrap_or(0);
                    if applied > 0 {
                        continue;
                    }
                    migration.apply(p).await?;
                }
            }
        }
        Ok(())
    }
}
```
